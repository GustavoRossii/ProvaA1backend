var builder = WebApplication.CreateBuilder(args);

// Adicione esta linha para configurar o CORS
builder.Services.AddCors(options =>
{
    options.AddPolicy("AllowReactApp",
        builder => builder
            .WithOrigins("http://localhost:3000") // Endereço do seu aplicativo React
            .AllowAnyMethod()
            .AllowAnyHeader()
            .AllowCredentials());
});

// ... (resto do seu código de configuração)

var app = builder.Build();

// Use esta linha antes de qualquer configuração de rota
app.UseCors("AllowReactApp");
